$$
\left(
\begin{array}{cc|cc}
a & b & 1 & 0 \\
c & d & 0 & 1
\end{array}
\right) =
\left(
\begin{array}{cc|cc}
1 & \frac{b}{a} & \frac{1}{a} & 0 \\
1 & \frac{d}{c} & 0 & \frac{1}{c}
\end{array}
\right) = \\
\left(
\begin{array}{cc|cc}
1 & \frac{b}{a} & \frac{1}{a} & 0 \\
0 & \frac{b}{a}-\frac{d}{c} & \frac{1}{a} & -\frac{1}{c}
\end{array}
\right) =
\left(
\begin{array}{cc|cc}
1 & \frac{b}{a} & \frac{1}{a} & 0 \\
0 & \frac{b}{a}-\frac{d}{c} & \frac{1}{a} & -\frac{1}{c}
\end{array}
\right)
